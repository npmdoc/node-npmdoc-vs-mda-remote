function vs-mda-remote.BuildLogger.prototype.begin (buildDir, logFileName, buildProcess)

  • description and source-code
    begin = function (buildDir, logFileName, buildProcess) {
    var pathToBuildLog = path.join(buildDir, logFileName);
    this.stream = fs.createWriteStream(pathToBuildLog);
    console.info('Opened build log file %s', pathToBuildLog);
    this.stream.on('error', function(err) {
        console.error('Error from writing to stream at %s: %s', pathToBuildLog, err);
    });
    var me = this;
    buildProcess.stdout.on('data', function(data) {
        me.log(data);
    });
    buildProcess.stderr.on('data', function(data) {
        me.log(data);
    });
}
